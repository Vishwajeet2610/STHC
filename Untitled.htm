    def __init__(self, name):
        Compound.__init__(self, name)  
    def Hig(self, T):  
        Tf = self.Tf
        Hf = self.Hf 
        Cp = self.CpIG 
        Hig = Hf + scint.quad(Cp, Tf, T)[0] 
        return Hig
